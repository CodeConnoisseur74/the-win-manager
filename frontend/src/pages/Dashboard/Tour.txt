// src/pages/Dashboard/Tour.js
import React, { useState } from 'react';
import Joyride, { STATUS } from 'react-joyride';

const steps = [
    {
        target: '.side-menu',
        content: 'This is your navigation menu. You can access different sections of the application from here.',
    },
    {
        target: '.filters-section',
        content: 'Use these filters to customize the data displayed in the chart and table.',
    },
    {
        target: '.wins-completion',
        content: 'This section shows the completion status of wins.',
    }
];

const Tour = () => {
    const [run, setRun] = useState(true);

    const handleJoyrideCallback = (data) => {
        const { status } = data;
        const finishedStatuses = [STATUS.FINISHED, STATUS.SKIPPED];
        if (finishedStatuses.includes(status)) {
            setRun(false);
        }
    };

    return (
        <Joyride
            steps={steps}
            run={run}
            continuous
            showSkipButton
            showProgress
            callback={handleJoyrideCallback}
        />
    );
};

export default Tour;
